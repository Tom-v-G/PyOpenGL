#version 330 core

layout (triangles) in;
layout (triangle_strip, max_vertices = 3) out;

in vec3 fragmentColor[];
out vec3 vFragColor;

void main() {
  for(int i = 0; i < 3; i++) { // You used triangles, so it's always 3
    vFragColor = fragmentColor[i];
    gl_Position = gl_in[i].gl_Position;
    EmitVertex();
  }
  EndPrimitive();
}
